a=int(input())
operator=input()
b=int(input())
match operator:
    case "+":
       print("the value of:",a,"+",b,"is:",a+b)
    case "-":
       print("the value of:",a,"-",b,"is:",a-b)
    case "*":
       print("the value of:",a,"*",b,"is:",a*b)
    case "/":
       print("the value of:",a,"/",b,"is:",a/b)
    case "//":
       print("the value of:",a,"//",b,"is:",a//b)
    case "%":
       print("the value of:",a,"%",b,"is:",a%b)
    case "_":
        print("invalid input")
