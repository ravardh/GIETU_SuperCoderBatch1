def partition(arr, p, r):
  pvt = arr[r]
  i = p - 1
  for j in range(p, r):
    if arr[j] <= pvt:
      i += 1
      arr[i], arr[j] = arr[j], arr[i]
  arr[i+1], arr[r] = arr[r], arr[i+1]
  return i + 1

def quicksort(arr, p, r):
  if p < r:
    q = partition(arr, p, r)
    quicksort(arr, p, q-1)
    quicksort(arr, q+1, r)

  return arr

arr = [32, 14, 44, 57, 12, 34, 89, 45]
n = len(arr)
quicksort(arr, 0, n - 1)
